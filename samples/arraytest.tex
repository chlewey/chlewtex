	\documentclass{article}
	\def\tempi{0}
	\def\tempii{1}
	\newcommand\mymacro[1]{%
	  \begingroup
	  \let\tempiii\tempi
	  \def\tempi{#1\tempiii}%
	  \def\tempii{#1}%
	  \mymacroi
	  \endgroup
	}
	\newcommand\mymacroi{%
	  \edef\tmp{\tempi&\tempii}%
	  \tmp
	}
	
	\begin{document}
	\renewcommand\mymacroi{%
	  \edef\tmp{\tempi,\tempii}%
	  \tmp
	}
	\[
	  \begin{array}{cccccc}
	    a & \mymacro{7} & b & \tempi & \tempii \\
	    \mymacro{2} & \mymacro{3} & \tempi & \tempii 
	  \end{array}
	\]
